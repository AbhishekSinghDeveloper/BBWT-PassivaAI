Typescript code

                export class ListsComponent implements OnInit {

                    themesOptions: any[];
                    selectedTheme: any;

                    sourceList: any[];
                    targetList: any[];

                    constructor() { }

                    ngOnInit() {
                        this.selectedTheme = {};
                        this.themesOptions = [];
                        this.themesOptions.push({ label: 'theme-blue', value: 'blue' });
                        this.themesOptions.push({ label: 'theme-blue-grey', value: 'blue-grey' });
                        this.themesOptions.push({ label: 'theme-brown', value: 'brown' });
                        this.themesOptions.push({ label: 'theme-cyan', value: 'cyan' });
                        this.themesOptions.push({ label: 'theme-dark-blue', value: 'dark-blue' });
                        this.themesOptions.push({ label: 'theme-dark-green', value: 'dark-green' });
                        this.themesOptions.push({ label: 'theme-green', value: 'green' });
                        this.themesOptions.push({ label: 'theme-grey', value: 'grey' });
                        this.themesOptions.push({ label: 'theme-indigo', value: 'indigo' });
                        this.themesOptions.push({ label: 'theme-purple-amber', value: 'purple-amber' });
                        this.themesOptions.push({ label: 'theme-purple-cyan', value: 'purple-cyan' });
                        this.themesOptions.push({ label: 'theme-teal', value: 'teal' });

                        this.sourceList = [];
                        this.sourceList.push({ model: 'Audi', year: 2017 });
                        this.sourceList.push({ model: 'VW', year: 2012 });
                        this.sourceList.push({ model: 'BMW', year: 2011 });

                        this.targetList = [];
                        this.targetList.push({ model: 'Opel', year: 2002 });
                        this.targetList.push({ model: 'Renault', year: 2013 });
                    }

                    changeTheme(event: any) {

                        console.log(event);
                        let themeLink: HTMLLinkElement = <HTMLLinkElement>document.getElementById('theme-css');
                        themeLink.href = '/theme-' + event.value + '.css';

                        let layoutLink: HTMLLinkElement = <HTMLLinkElement>document.getElementById('layout-css');
                        layoutLink.href = '/layout-' + event.value + '.css';
                        event.preventDefault();
                    }
                }
